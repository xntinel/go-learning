# 7. XML Encoding and Decoding

XML is not JSON with angle brackets. Attributes, character data, element names, and validation rules all need explicit modeling. This lesson builds an `rssxml` library that parses and writes a small RSS feed with `encoding/xml`, validates decoded data, and exposes safe accessors.

## Concepts

### Tags Describe XML Shape

`encoding/xml` maps exported struct fields through tags such as `xml:"version,attr"`, `xml:",chardata"`, and `xml:"channel"`. An `XMLName xml.Name` field can pin the element name for the root struct.

### Parsing Is Not Domain Validation

`xml.Unmarshal` can successfully parse a feed that is meaningless to your application. A library should validate required fields after decoding and return wrapped sentinel errors for invalid domain data.

### Generated XML Should Reparse

`xml.MarshalIndent` writes XML elements, but it does not prepend `xml.Header`. If the output is intended as a standalone document, prepend the header and test that your own parser accepts the generated bytes.

### Exercise 1: Model and Validate an RSS Feed

Create `feed.go`:

```go
package rssxml

import (
	"encoding/xml"
	"errors"
	"fmt"
)

var ErrInvalidFeed = errors.New("invalid rss feed")

type Feed struct {
	XMLName      xml.Name `xml:"rss"`
	VersionValue string   `xml:"version,attr"`
	Channel      Channel  `xml:"channel"`
}

type Channel struct {
	TitleValue       string `xml:"title"`
	LinkValue        string `xml:"link"`
	DescriptionValue string `xml:"description"`
	ItemsValue       []Item `xml:"item"`
}

type Item struct {
	TitleValue       string `xml:"title"`
	LinkValue        string `xml:"link"`
	DescriptionValue string `xml:"description"`
	GUIDValue        GUID   `xml:"guid"`
}

type GUID struct {
	PermaLink bool   `xml:"isPermaLink,attr"`
	Value     string `xml:",chardata"`
}

func NewFeed(title, link, description string, items []Item) Feed {
	return Feed{VersionValue: "2.0", Channel: Channel{TitleValue: title, LinkValue: link, DescriptionValue: description, ItemsValue: append([]Item(nil), items...)}}
}

func NewItem(title, link, description, guid string, permaLink bool) Item {
	return Item{TitleValue: title, LinkValue: link, DescriptionValue: description, GUIDValue: GUID{PermaLink: permaLink, Value: guid}}
}

func Parse(data []byte) (Feed, error) {
	var feed Feed
	if err := xml.Unmarshal(data, &feed); err != nil {
		return Feed{}, fmt.Errorf("%w: decode xml: %v", ErrInvalidFeed, err)
	}
	if err := feed.validate(); err != nil {
		return Feed{}, err
	}
	return feed, nil
}

func Encode(feed Feed) ([]byte, error) {
	if err := feed.validate(); err != nil {
		return nil, err
	}
	data, err := xml.MarshalIndent(feed, "", "\t")
	if err != nil {
		return nil, fmt.Errorf("%w: encode xml: %v", ErrInvalidFeed, err)
	}
	out := make([]byte, 0, len(xml.Header)+len(data))
	out = append(out, xml.Header...)
	out = append(out, data...)
	return out, nil
}

func (f Feed) Version() string       { return f.VersionValue }
func (f Feed) Title() string         { return f.Channel.TitleValue }
func (f Feed) Link() string          { return f.Channel.LinkValue }
func (f Feed) Description() string   { return f.Channel.DescriptionValue }
func (f Feed) Items() []Item         { return append([]Item(nil), f.Channel.ItemsValue...) }
func (i Item) Title() string         { return i.TitleValue }
func (i Item) Link() string          { return i.LinkValue }
func (i Item) Description() string   { return i.DescriptionValue }
func (i Item) GUID() string          { return i.GUIDValue.Value }
func (i Item) GUIDIsPermaLink() bool { return i.GUIDValue.PermaLink }

func (f Feed) validate() error {
	if f.VersionValue != "2.0" {
		return fmt.Errorf("%w: version must be 2.0", ErrInvalidFeed)
	}
	if f.Channel.TitleValue == "" {
		return fmt.Errorf("%w: channel title is required", ErrInvalidFeed)
	}
	if len(f.Channel.ItemsValue) == 0 {
		return fmt.Errorf("%w: at least one item is required", ErrInvalidFeed)
	}
	for _, item := range f.Channel.ItemsValue {
		if item.TitleValue == "" {
			return fmt.Errorf("%w: item title is required", ErrInvalidFeed)
		}
		if item.GUIDValue.Value == "" {
			return fmt.Errorf("%w: item guid is required", ErrInvalidFeed)
		}
	}
	return nil
}
```

### Exercise 2: Test Decode, Encode, and Reparse

Create `feed_test.go`:

```go
package rssxml

import (
	"errors"
	"fmt"
	"strings"
	"testing"
)

func TestParse(t *testing.T) {
	t.Parallel()

	tests := []struct {
		name      string
		xml       string
		wantTitle string
		wantItems int
		wantErr   error
	}{
		{name: "valid feed", xml: `<rss version="2.0"><channel><title>Go News</title><link>https://go.dev</link><description>Updates</description><item><title>Release</title><link>https://go.dev/doc</link><description>Notes</description><guid isPermaLink="true">https://go.dev/doc</guid></item></channel></rss>`, wantTitle: "Go News", wantItems: 1},
		{name: "wrong version", xml: `<rss version="1.0"><channel><title>Go News</title><item><title>Release</title><guid>x</guid></item></channel></rss>`, wantErr: ErrInvalidFeed},
		{name: "bad xml", xml: `<rss version="2.0"><channel>`, wantErr: ErrInvalidFeed},
	}
	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			t.Parallel()
			got, err := Parse([]byte(tt.xml))
			if tt.wantErr != nil {
				if !errors.Is(err, tt.wantErr) {
					t.Fatalf("Parse() error = %v, want %v", err, tt.wantErr)
				}
				return
			}
			if err != nil {
				t.Fatal(err)
			}
			if got.Title() != tt.wantTitle || len(got.Items()) != tt.wantItems {
				t.Fatalf("feed title=%q items=%d", got.Title(), len(got.Items()))
			}
		})
	}
}

func TestEncode(t *testing.T) {
	t.Parallel()

	tests := []struct {
		name    string
		feed    Feed
		wantErr error
	}{
		{name: "valid feed", feed: NewFeed("Go News", "https://go.dev", "Updates", []Item{NewItem("Release", "https://go.dev/doc", "Notes", "https://go.dev/doc", true)})},
		{name: "missing item", feed: NewFeed("Go News", "https://go.dev", "Updates", nil), wantErr: ErrInvalidFeed},
	}
	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			t.Parallel()
			got, err := Encode(tt.feed)
			if tt.wantErr != nil {
				if !errors.Is(err, tt.wantErr) {
					t.Fatalf("Encode() error = %v, want %v", err, tt.wantErr)
				}
				return
			}
			if err != nil {
				t.Fatal(err)
			}
			if !strings.HasPrefix(string(got), `<?xml version="1.0" encoding="UTF-8"?>`) {
				t.Fatalf("Encode() missing XML header: %q", got)
			}
			if _, err := Parse(got); err != nil {
				t.Fatalf("generated XML did not reparse: %v", err)
			}
		})
	}
}

func ExampleParse() {
	feed, _ := Parse([]byte(`<rss version="2.0"><channel><title>Go News</title><link>https://go.dev</link><description>Updates</description><item><title>Release</title><link>https://go.dev/doc</link><description>Notes</description><guid isPermaLink="true">https://go.dev/doc</guid></item></channel></rss>`))
	fmt.Println(feed.Title())
	fmt.Println(feed.Items()[0].Title())
	// Output:
	// Go News
	// Release
}
```

package main

import (
	"fmt"

	"example.com/rssxml"

func main() {

	if err != nil {
		log.Fatal(err)
	}
	parsed, err := rssxml.Parse(data)
	if err != nil {
		log.Fatal(err)
	}
	fmt.Printf("%s has %d item\n", parsed.Title(), len(parsed.Items()))
}

## Common Mistakes

- Wrong: assuming successful XML parsing means a valid feed. What happens: missing titles or items pass through. Fix: validate after decoding.
- Wrong: returning internal slices directly. What happens: callers can mutate library state. Fix: return copies from accessors.
- Wrong: forgetting `xml.Header` for standalone documents. What happens: the XML may still parse, but the generated document is incomplete for many consumers. Fix: prepend `xml.Header` when encoding a document.

## Summary

- XML tags model elements, attributes, and character data.
- `xml.Unmarshal` parses syntax; your package validates domain rules.
- `xml.MarshalIndent` can generate readable XML, and `xml.Header` makes it a standalone document.
- XML libraries should test both parse and encode paths.
